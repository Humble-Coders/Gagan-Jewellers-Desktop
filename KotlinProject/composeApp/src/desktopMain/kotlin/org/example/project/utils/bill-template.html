package org.example.project.utils import org.example.project.data.Order import org.example.project.data.User import org.example.project.data.CartItem import java.io.FileOutputStream import java.nio.file.Path import java.text.SimpleDateFormat import java.util.* import org.xhtmlrenderer.pdf.ITextRenderer class HtmlToPdfBillGenerator { fun generatePdfBill(order: Order, customer: User, outputPath: Path): Boolean { return try { val outputFile = outputPath.toFile() outputFile.parentFile?.mkdirs() val htmlContent = generatePdfCompatibleHtml(order, customer) val renderer = ITextRenderer() renderer.setDocumentFromString(htmlContent) renderer.layout() FileOutputStream(outputFile).use { outputStream -> renderer.createPDF(outputStream) } outputFile.exists() && outputFile.length() > 0 } catch (e: Exception) { e.printStackTrace() false } } private fun generatePdfCompatibleHtml(order: Order, customer: User): String { val dateFormat = SimpleDateFormat("dd/MM/yyyy HH:mm", Locale.getDefault()) val orderDate = dateFormat.format(Date(order.timestamp)) val subtotal = order.subtotal val gstAmount = order.gstAmount val discountAmount = order.discountAmount val totalAmount = order.totalAmount val gstPercentage = if (subtotal > 0) ((gstAmount / subtotal) * 100).toInt() else 0 return """ Invoice ${escapeXml(order.id)}

Vishal Gems and Jewels Pvt. Ltd.

Premium Gold and Silver Jewelry

INVOICE

No: ${escapeXml(order.id)}

Date: ${escapeXml(orderDate)}

Payment: ${escapeXml(order.paymentMethod.name)}

${escapeXml(customer.name)} ${if (customer.phone.isNotEmpty()) " | Phone: ${escapeXml(customer.phone)}" else ""} ${if (customer.email.isNotEmpty()) " | Email: ${escapeXml(customer.email)}" else ""}
${generateItemRows(order.items)}
Item Material Wt(g) Qty Rate Making Stone VA Amount
${if (gstAmount > 0) "" else ""} ${if (discountAmount > 0) "" else ""}
Subtotal:Rs. ${String.format("%.2f", subtotal)}
GST (${gstPercentage}%):Rs. ${String.format("%.2f", gstAmount)}
Discount:-Rs. ${String.format("%.2f", discountAmount)}
TOTAL:Rs. ${String.format("%.2f", totalAmount)}

Thank you for your business!

""".trimIndent() } private fun generateItemRows(items: List): String { return items.joinToString("") { item -> val weight = if (item.selectedWeight > 0) item.selectedWeight else parseWeight(item.product.weight) val rate = getMetalRate(item) val baseAmount = weight * item.quantity * rate // Get charges from product or use defaults val makingCharges = item.product.makingCharges ?: 0.0 val stoneCharges = item.product.stoneCharges ?: 0.0 val vaCharges = item.product.vaCharges ?: 0.0 val totalAmount = baseAmount + makingCharges + stoneCharges + vaCharges """ ${escapeXml(item.product.name)} ${escapeXml(item.product.materialType)} ${String.format("%.2f", weight)} ${item.quantity} ${String.format("%.0f", rate)} ${String.format("%.0f", makingCharges)} ${String.format("%.0f", stoneCharges)} ${String.format("%.0f", vaCharges)} ${String.format("%.2f", totalAmount)} """ } } private fun escapeXml(text: String): String { return text .replace("&", "&") .replace("<", "<") .replace(">", ">") .replace("\"", """) .replace("'", "'") } private fun parseWeight(weightStr: String): Double { return weightStr.replace(Regex("[^0-9.]"), "").toDoubleOrNull() ?: 0.0 } private fun getMetalRate(cartItem: CartItem): Double { return when { cartItem.product.materialType.contains("gold", ignoreCase = true) -> 5500.0 cartItem.product.materialType.contains("silver", ignoreCase = true) -> 75.0 else -> 5500.0 } } private fun getPdfCompatibleStyles(): String { return """ * { margin: 0; padding: 0; box-sizing: border-box; } body { font-family: Arial, sans-serif; font-size: 9pt; color: #333; } .container { padding: 15px; } .header { display: table; width: 100%; border-bottom: 2px solid #B8973D; padding-bottom: 8px; margin-bottom: 10px; } .company { display: table-cell; width: 60%; vertical-align: top; } .company h1 { color: #B8973D; font-size: 18pt; margin-bottom: 2px; } .tagline { color: #666; font-size: 8pt; } .invoice-meta { display: table-cell; width: 40%; text-align: right; vertical-align: top; } .invoice-meta h2 { color: #B8973D; font-size: 16pt; margin-bottom: 4px; } .invoice-meta p { font-size: 8pt; margin-bottom: 2px; } .customer { padding: 6px 8px; background: #F9F9F9; border-left: 3px solid #B8973D; margin-bottom: 10px; font-size: 8pt; } .items { width: 100%; border-collapse: collapse; margin-bottom: 10px; font-size: 8pt; } .items th { background: #B8973D; color: white; padding: 5px 4px; text-align: left; font-size: 7pt; } .items td { padding: 4px; border-bottom: 1px solid #DDD; } .items tbody tr:nth-child(even) { background: #FAFAFA; } .summary { margin-bottom: 8px; } .summary-table { float: right; width: 200px; border-collapse: collapse; font-size: 9pt; } .summary-table td { padding: 4px 8px; border-bottom: 1px solid #EEE; } .summary-table td:last-child { text-align: right; font-weight: bold; } .discount-row { color: #4CAF50; } .total-row { border-top: 2px solid #B8973D; color: #B8973D; font-size: 10pt; } .footer { clear: both; text-align: center; padding-top: 8px; border-top: 1px solid #B8973D; font-size: 8pt; color: #B8973D; } """.trimIndent() }}
